    16     public GraphGUI(TemperatureModel model, int h, int v) {

    18         this.model = model;
    19         _farenheit = new TemperatureGauge(-200, 300);
    20         Panel Top = new Panel();
    21         add("North", Top);

    23         gauges.setSize(500, 1500); // This size seems very large, might be a typo in the

    34     }

    36     @Override
    37     public void paint(Graphics g) {
    38         int farenheit = (int) model.getC();
    39         _farenheit.set(farenheit);
    40         gauges.repaint();
    41         super.paint(g);
    42     }
    43
    44     class TemperatureCanvas extends Canvas {
    45         private TemperatureGauge _farenheit;
    46         private static final int width = 20;
    47         private static final int top = 20;
    48         private static final int left = 100;
    49         private static final int right = 250; // Not used
    50         private static final int height = 150;
    51
    52         public TemperatureCanvas(TemperatureGauge farenheit) {
    53             _farenheit = farenheit;
    54         }
    55
    56         @Override
    57         public void paint(Graphics g) {
    58             g.setColor(Color.black);
    59             g.drawRect(left, top, width, height);
    60             g.setColor(Color.red);
    61             g.fillOval(left + width / 2, top + height - width / 3, width * 2, width * 2); //
       Thermometer bulb
    62             g.setColor(Color.black);
    63             g.drawOval(left + width / 2, top + height - width / 3, width * 2, width * 2); //
       Thermometer bulb outline
    64             g.setColor(Color.white);
    65             g.fillRect(left + 1, top + 1, width - 1, height - 1); // Clear inside of
       thermometer
    66             g.setColor(Color.red);
    67             long redtop = height * (_farenheit.get() - _farenheit.getMax()) /
       (_farenheit.getMin() - _farenheit.getMax());
    68             g.fillRect(left + 1, top + (int) redtop, width - 1, height - (int) redtop); //
       Fill with red based on temperature
    69         }
    70     }
